def get_cust_dim(log, spark, rds_schema_name):
    log.info("Started selecting cust_dim from %s.", rds_schema_name)
    sql_query = f"""
        SELECT cust_id,
               trade_chanl_curr_id AS trade_chanl_id
        FROM {src_catalog_name}.{rds_schema_name}.cust_dim
        WHERE curr_ind = 'Y'
    """
    df = spark.sql(sql_query)
    log.info("Selecting cust_dim from %s has finished.", rds_schema_name)
    return df

def get_trade_chanl_hier_dim(log, spark, rds_schema_name):
    log.info("Started selecting trade_chanl_hier_dim from %s.", rds_schema_name)
    sql_query = f"""
        SELECT trade_chanl_2_long_name AS channel_name,
               trade_chanl_7_id AS trade_chanl_id
        FROM {src_catalog_name}.{rds_schema_name}.trade_chanl_hier_dim
        WHERE curr_ind = 'Y'
          AND trade_chanl_hier_id = '658'
    """
    df = spark.sql(sql_query)
    log.info("Selecting trade_chanl_hier_dim from %s has finished.", rds_schema_name)
    return df


def manageOutput(log, spark, df, cache_ind, df_name, schema_name):
    """
    Simplified manageOutput function.
    When cache_ind == 1, registers a temporary view and caches the DataFrame.
    """
    if cache_ind == 1:
        temp_view = df_name + DEBUG_SUFFIX.upper() + "_CACHE_VW"
        df.createOrReplaceTempView(temp_view)
        spark.sql(f"CACHE TABLE {temp_view}")
        log.info("Data frame '%s' cached as %s.", df_name, temp_view)
    return

def removeDebugTables(log, spark, schema_name):
    """
    Remove any tables in schema_name that end with the DEBUG_SUFFIX.
    """
    debug_postfix = DEBUG_SUFFIX.upper()
    spark.sql(f"USE {schema_name}")
    tables = spark.sql("SHOW TABLES")
    log.info("Started dropping DEBUG tables.")
    for row in tables.collect():
        if row.tableName.upper().endswith(debug_postfix):
            spark.sql(f"DROP TABLE IF EXISTS {row.database}.{row.tableName}")
            log.info("Dropped table %s.%s", row.database, row.tableName)
    return

def load_on_time_data_hub_star(spark, log, schema_name, target_table):
    """
    Refreshes required tables, calls the processing function,
    and inserts the resulting DataFrame into the target table.
    """
    spark.sql(f"REFRESH TABLE {src_catalog_name}.{rds_schema_name}.cust_dim")
    spark.sql(f"REFRESH TABLE {src_catalog_name}.{rds_schema_name}.trade_chanl_hier_dim")
    log.info("Started loading {}.{} table.".format(schema_name, target_table))
    
    df = get_on_time_data_hub_star(log, spark)
    # If the target table already exists, the new data will overwrite the old data.
    df.write.insertInto(f"{schema_name}.{target_table}", overwrite=True)
    log.info("Loading {}.{} table has finished.".format(schema_name, target_table))
    return
